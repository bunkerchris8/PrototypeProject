#include <Arduino.h>
#include <Stepper.h>

const int stepsPerRevolution = 2038;

//This uses D, not GPIO

Stepper myStepper(stepsPerRevolution, 8, 9, 10, 11);

void setup() {
  myStepper.setSpeed(5);
}

void loop() {
  myStepper.step(stepsPerRevolution);
  delay(1000);
  myStepper.step(-stepsPerRevolution);
  delay(1000);
}
